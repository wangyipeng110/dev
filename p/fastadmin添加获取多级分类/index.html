    public function qdtype1(){
        $list= $this->model->where('remark',1)->select();
        foreach ($list as $k=>$v){
            $v['name'] = $v['title'];
            $v['value'] =$v['id'];
        }
        $this->success('','',$list);
    }
    public function qdtype2(){
        $pid = $this ->request->get('pid');
        
        $list= $this->model->where('remark',2)->where('pid',$pid)->select();
        foreach ($list as $k=>$v){
            $v['name'] = $v['title'];
            $v['value'] =$v['id'];
        }
        $this->success('','',$list);
    }
